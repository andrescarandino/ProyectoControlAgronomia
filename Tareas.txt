                             CALCULOS RESUMENES Y RECOPILACION DE DATOS


COSTO DIRECTO = TOTAL INSUMOS + TOTAL LABORES      // usd/ha

RENDIMIENTO = QUINTALES/HA

PRECIO = RENDIMIENTO / 10

GASTOS COMERCIALIZACION = RENDIMIENTO * %COMERZIALIZACION (transporte + comision)
                           ejemplo = 25,5usd/qq * 17% ---->  25,5*0.17

PRECIO NETO = PRECIO - GASTOS COMERCIALIZACION

INGRESO BRUTO = PRECIO NETO * RENDIMIENTO

MB = INGRESO BRUTO - COSTO DIRECTO

RTO INDIFERENCIA = COSTO DIRECTO / PRECIO NETO

MB/CD = MB /  COSTO DIRECTO
